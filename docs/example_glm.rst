    class PoissonRegression(pf.Model):

        def __init__(self, dims):
            self.w = pf.Parameter([dims, 1])
            self.b = pf.Parameter([1, 1])

        def __call__(self, x):
            return pf.Poisson(tf.exp(x @ self.w() + self.b()))
